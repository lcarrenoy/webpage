      title: `Histograma con ${nPoints} puntos`,
      width: 640,
      height: 400,
      color: {legend: true},
      marks: [
        Plot.rectY(filteredData, 
          Plot.binX({y: "count"}, {
            x: "x", 
            fill: "category",
            stroke: "white"
          })
        )
      ]
    });
  } else {
    return Plot.plot({
      title: `Box Plot con ${nPoints} puntos`,

      height: 400,

      marks: [
        Plot.boxY(filteredData, {
          x: "category", 
          y: "y", 
          fill: "category"
        })
      ]
    });
  }
}
